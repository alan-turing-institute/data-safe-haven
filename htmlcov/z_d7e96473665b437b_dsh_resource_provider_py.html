20 old_props: dict[str, Any], 

21 new_props: dict[str, Any], 

22 excluded_props: Sequence[str] = [], 

23 ) -> DiffResult: 

24 """ 

25 Calculate a diff between an old and new set of props, excluding any if requested. 

26 

27 Args: 

28 old_props: the outputs from the last create operation 

29 new_props: a set of checked inputs 

30 excluded_props: any props to exclude from the comparison 

31 

32 Returns: 

33 DiffResult: 

34 - changes: whether any non-excluded props have changed 

35 - replaces: any non-excluded props that have changed 

36 - stables: any properties that are excluded or unchanged 

37 - delete_before_replace: True 

38 """ 

39 # List any values that were not present in old_props or have been changed 

40 # Exclude any from excluded_props which should not trigger a diff 

41 altered_props = [ 

42 property_ 

43 for property_ in [ 

44 key for key in new_props.keys() if key not in excluded_props 

45 ] 

46 if (property_ not in old_props) 

47 or (old_props[property_] != new_props[property_]) 

48 ] 

49 stable_props = [ 

50 property_ 

51 for property_ in old_props.keys() 

52 if property_ not in altered_props 

53 ] 

54 return DiffResult( 

55 changes=(altered_props != []), # changes are needed 

56 replaces=altered_props, # properties that cannot be updated in-place 

57 stables=stable_props, # properties that will not change on update 

58 delete_before_replace=True, # delete the existing resource before replacing 

59 )

62 self, old_props: dict[str, Any], new_props: dict[str, Any] 

63 ) -> CheckResult: 

64 """ 

65 Invoked on update before any other method. 

66 Verify that the props are valid or return useful error messages if they are not.
